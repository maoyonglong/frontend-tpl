render(objopt) → {Tpl}

Source:
  • index.js, line 71

The method to render the ast

Example
// you can use variable to render html string via "{{}}" and data
var tpl = new Tpl()
 var data = {
     divClass: 'div',
     out: 'outer'
 }
 var htmlString = `
      <div class="{{ divClass }}">
          <p>{{ "inner" }}</p>
      </div>
      <p>{{ out + "p" }}</p>
   `
 var dom = tpl.parse(htmlString).render({ data: data }).getDom()
 //
 // dom is:
 // [
 //  <div class="div">
 //      <p>inner</p>
 //  </div>,
 //  <p>outerp</p>
 // ]
 //
